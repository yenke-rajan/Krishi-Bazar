router.delete('/:id', verifyToken, async (req, res) => {
  try {
    const orderId = req.params.id;
    // Use whatever field name the JWT payload uses for user id
    // Check your verifyToken middleware — it might be req.user.id 
    // or req.user.userId or req.user.sub
    const userId = (req as any).user.id;  
    // ADJUST THIS LINE based on what you find in middleware/auth.ts

    // Step 1: Find the order
    const order = db.prepare(
      'SELECT * FROM orders WHERE id = ?'
    ).get(orderId) as any;

    if (!order) {
      return res.status(404).json({ error: 'Order not found' });
    }

    // Step 2: Verify ownership
    if (order.client_id !== userId) {
      return res.status(403).json({ 
        error: 'You do not have permission to cancel this order' 
      });
    }

    // Step 3: Check if cancellable
    // SUPPLY (farmer): only cancellable at ORDER_RECEIVED
    // DEMAND (wholesaler): only cancellable at ORDER_RECEIVED
    const cancellableStatus = 'ORDER_RECEIVED';
    if (order.status !== cancellableStatus) {
      return res.status(400).json({ 
        error: 'Order cannot be cancelled — it is already being processed',
        error_np: 'अर्डर रद्द गर्न सकिँदैन — यो पहिले नै प्रक्रियामा छ'
      });
    }

    // Step 4: Delete the order
    db.prepare('DELETE FROM orders WHERE id = ?').run(orderId);

    return res.status(200).json({ 
      message: 'Order cancelled successfully',
      message_np: 'अर्डर सफलतापूर्वक रद्द गरियो'
    });

  } catch (err) {
    console.error('[DELETE /orders/:id] error:', err);
    return res.status(500).json({ error: 'Server error', details: String(err) });
  }
});

FIX 2 — STOCK SUMMARY SHOWS NO NUMBERS
═══════════════════════════════════

The inventory summary is returning empty or zero values.
This is a SQL query problem. Rewrite the endpoint completely.

In server/src/routes/inventory.ts, rewrite GET /summary:

Read schema.ts first to confirm:
  - The table is named 'orders' (not 'order')
  - The catalog table is named 'catalog'
  - The inventory table is named 'inventory_ledger'
  - Column names: orders.layer_type, orders.status, orders.weight_kg,
                  orders.crop_id, orders.client_id
  - inventory_ledger columns: crop_id, delta_quantity, tracking_type

Then write this as THREE separate queries joined in JavaScript
(this avoids complex multi-join SQL bugs with SQLite):

router.get('/summary', verifyToken, (req, res) => {
  try {
    // Query 1: Get all catalog items
    const catalogItems = db.prepare(
      `SELECT id, crop_name, crop_name_np, category, is_available 
       FROM catalog 
       ORDER BY crop_name ASC`
    ).all() as any[];

    // Query 2: Pending supply orders (farmer orders not yet collected)
    // These are KG "to receive" from farmers
    const toReceiveRows = db.prepare(
      `SELECT crop_id, SUM(weight_kg) as total
       FROM orders
       WHERE layer_type = 'SUPPLY'
         AND status IN ('ORDER_RECEIVED', 'DISPATCHED_TO_COLLECT')
       GROUP BY crop_id`
    ).all() as any[];

    // Query 3: Pending demand orders (wholesaler orders not yet delivered)
    // These are KG "to deliver" to wholesalers
    const toDeliverRows = db.prepare(
      `SELECT crop_id, SUM(weight_kg) as total
       FROM orders
       WHERE layer_type = 'DEMAND'
         AND status IN ('ORDER_RECEIVED', 'DISPATCHED')
       GROUP BY crop_id`
    ).all() as any[];

    // Query 4: Physical warehouse stock from inventory_ledger
    // in_warehouse = total received - total delivered (from completed orders)
    const warehouseRows = db.prepare(
      `SELECT 
         crop_id,
         SUM(CASE WHEN tracking_type = 'RECEIVED' 
             THEN delta_quantity ELSE 0 END) as total_received,
         SUM(CASE WHEN tracking_type = 'DELIVERED' 
             THEN delta_quantity ELSE 0 END) as total_delivered,
         MAX(recorded_at) as last_updated,
         (SELECT tracking_type FROM inventory_ledger il2 
          WHERE il2.crop_id = inventory_ledger.crop_id 
          ORDER BY recorded_at DESC LIMIT 1) as last_entry_type
       FROM inventory_ledger
       GROUP BY crop_id`
    ).all() as any[];

    // Build lookup maps for O(1) access
    const toReceiveMap: Record<string, number> = {};
    toReceiveRows.forEach(r => { 
      toReceiveMap[r.crop_id] = Number(r.total) || 0; 
    });

    const toDeliverMap: Record<string, number> = {};
    toDeliverRows.forEach(r => { 
      toDeliverMap[r.crop_id] = Number(r.total) || 0; 
    });

    const warehouseMap: Record<string, { 
      in_warehouse: number; 
      last_updated: number | null;
      last_entry_type: string | null;
    }> = {};
    warehouseRows.forEach(r => {
      warehouseMap[r.crop_id] = {
        in_warehouse: (Number(r.total_received) || 0) - 
                      (Number(r.total_delivered) || 0),
        last_updated: r.last_updated || null,
        last_entry_type: r.last_entry_type || null,
      };
    });

    // Combine into final summary
    const summary = catalogItems.map(item => {
      const to_receive = toReceiveMap[item.id] || 0;
      const to_deliver = toDeliverMap[item.id] || 0;
      const warehouse = warehouseMap[item.id];
      const in_warehouse = warehouse?.in_warehouse || 0;
      // needed: positive = shortfall (we owe more than we have)
      //         negative = surplus (we have more than needed)
      const needed = to_deliver - in_warehouse;

      return {
        crop_id: item.id,
        crop_name: item.crop_name,
        crop_name_np: item.crop_name_np,
        category: item.category,
        to_receive,
        to_deliver,
        in_warehouse,
        needed,
        last_updated: warehouse?.last_updated || null,
        last_entry_type: warehouse?.last_entry_type || null,
      };
    });

    console.log('[GET /inventory/summary] returning', summary.length, 'items');
    console.log('[GET /inventory/summary] sample:', JSON.stringify(summary[0]));

    return res.json({ summary });

  } catch (err) {
    console.error('[GET /inventory/summary] error:', err);
    return res.status(500).json({ error: 'Server error', details: String(err) });
  }
});
